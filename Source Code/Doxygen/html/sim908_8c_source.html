685 void _SIM908_callback(char data) {
686  _rx_response_length++;
687  _rx_buffer[_rx_buffer_tail = (_rx_buffer_tail + 1) % RX_BUFFER_SIZE] = data; /* Stores received data in buffer. This technically starts from index '1', but as the buffer is circular, it does not matter */
688 
689  if (data == CR) { /* Checking and counting for CR and LF */
690  _CR_counter++;
691  } else if (data == LF) {
692  _LF_counter++;
693  }
694 
695  if (_CR_counter > 0 && _LF_counter > 0) {
696  _CR_counter = _LF_counter = 0;
697  if (_rx_response_length == 2 &&
698  (_check_response(SIM908_RESPONSE_CR_LF) ||
699  _check_response(SIM908_RESPONSE_LF_CR))) { /* Skipping empty lines */
700  } else
701  if (_rx_response_length == 4 &&
702  _check_response(SIM908_RESPONSE_OK)) { /* OK */
703  _ack_response_flag = SIM908_FLAG_OK;
704  } else
705  if (_rx_response_length == 7 &&
706  _check_response(SIM908_RESPONSE_ERROR)) { /* Error */
707  _ack_response_flag = SIM908_FLAG_ERROR;
708  } else
709  if (_gps_pull_flag == SIM908_FLAG_GPS_PULL &&
710  _check_response(SIM908_RESPONSE_GPS_PULL)) { /* GPS pull */
711  _gps_response_tail = _rx_buffer_tail;
712  _gps_response_length = _rx_response_length;
713  _ack_gps_response_flag = SIM908_FLAG_GPS_PULL_OK;
714  _gps_pull_flag = SIM908_FLAG_WAITING;
715  } else
716  if (_rx_response_length == 10 &&
717  _check_response(SIM908_RESPONSE_CREG)) { /* CREG */
718  EXT_CONNECTION_CREG_FLAG = _char_at(7, _rx_buffer_tail, _rx_response_length) - '0'; /* Subtracting '0' (0x30) to get the value as an integer */
719  } else
720  if (_rx_response_length == 11 &&
721  _check_response(SIM908_RESPONSE_GPS_READY)) { /* GPS Ready */
722  _ack_gps_response_flag = SIM908_FLAG_GPS_OK;
723  } else
724  if (_ftp_sending_flag == SIM908_FLAG_FTP_SENDING &&
725  _check_response(SIM908_RESPONSE_FTP_PUT)) { /* FTPPUT */
726  /* FTP PUT OPEN SESSION: "+FTPPUT:1,1,1260"
727  FTP PUT RESPONSE: "+FTPPUT:2,140"
728  FTP PUT CLOSE SESSION: "+FTPPUT:1,0" */
729  char c1 = _char_at(8, _rx_buffer_tail, _rx_response_length);
730  char c2 = _char_at(10, _rx_buffer_tail, _rx_response_length);
731  char c3 = _char_at(11, _rx_buffer_tail, _rx_response_length);
732  if (c1 == '1' && c2 == '1' && c3 == ',') {
733  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_OPEN;
734  } else if(c1 == '2' && c2 == '1' && c3 == '4') {
735  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_SUCCESS;
736  } else if(c1 == '1' && c2 == '0') {
737  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_CLOSE;
738  } else {
739  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_ERROR;
