732  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_SUCCESS;
733  #ifdef DEBUG_SIM908_CALLBACK

735  #endif
736  } else if((c1 == '1') && (c2 == '0')) {
737  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_CLOSE;

743  #ifdef DEBUG_SIM908_CALLBACK

745  #endif
746  }
747  } else
748  if ((_gps_pull_flag == SIM908_FLAG_GPS_PULL) &&
749  (_check_response(SIM908_RESPONSE_GPS_PULL))) { /* GPS pull */
750  _gps_response_tail = _rx_buffer_tail;
751  _gps_response_length = _rx_response_length;
752  _ack_gps_response_flag = SIM908_FLAG_GPS_PULL_OK;
753  _gps_pull_flag = SIM908_FLAG_WAITING;
754  #ifdef DEBUG_SIM908_CALLBACK
755  uart1_send_string("rx:> gps pull\r\n");
756  #endif
757  } else
758  if ((_rx_response_length == 4U) &&
759  (_check_response(SIM908_RESPONSE_OK) == true)) { /* OK */
760  _ack_response_flag = SIM908_FLAG_OK;
761  #ifdef DEBUG_SIM908_CALLBACK
762  uart1_send_string("rx:> ok\r\n");
763  #endif
764  } else
765  if ((_rx_response_length == 7U) &&
766  (_check_response(SIM908_RESPONSE_ERROR) == true)) { /* Error */
767  _ack_response_flag = SIM908_FLAG_ERROR;
768  #ifdef DEBUG_SIM908_CALLBACK
769  uart1_send_string("rx:> error\r\n");
770  #endif
771  } else
772  if (((_rx_response_length == 10U) ||
773  (_rx_response_length == 12U)) &&
774  (_check_response(SIM908_RESPONSE_CREG) == true)) { /* CREG */
775  EXT_CONNECTION_CREG_FLAG = _char_at(_rx_response_length - 3U, _rx_buffer_tail, _rx_response_length) - '0'; /* Subtracting '0' (0x30) to get the value as an integer */
776  #ifdef DEBUG_SIM908_CALLBACK
777  uart1_send_string("rx:> creg\r\n");
778  #endif
779  } else
780  if ((_rx_response_length == 11U) &&
781  (_check_response(SIM908_RESPONSE_GPS_READY) == true)) { /* GPS Ready */
782  _ack_gps_response_flag = SIM908_FLAG_GPS_OK;
783  #ifdef DEBUG_SIM908_CALLBACK
784  uart1_send_string("rx:> gps ready\r\n");
785  #endif
786  } else
787  if ((_system_running_flag == SIM908_FLAG_WAITING) &&
788  (_rx_response_length == 5U) &&
789  (_check_response(SIM908_RESPONSE_RDY) == true)) { /* System ready */
790  _system_running_flag = SIM908_FLAG_RUNNING;
791  #ifdef DEBUG_SIM908_CALLBACK
792  uart1_send_string("rx:> rdy\r\n");
793  #endif
794  }
795  else if ((_rx_response_length == 4U) &&
796  (_check_response(SIM908_RESPONSE_AT) == true)) { /* Sync AT cmd */
797  _rx_response_length = 0U;
798  #ifdef DEBUG_SIM908_CALLBACK
799  uart1_send_string("rx:> at\r\n");
800  #endif
801  } else {
802  #ifdef DEBUG_SIM908_CALLBACK_NOT_CAUGHT
803  uart1_send_string("rx:> NOT CAUGHT: >>");
804  for (uint8_t i = 0U; i < _rx_response_length; i++) {
805  uart1_send_char(_char_at(i, _rx_buffer_tail, _rx_response_length));
806  }
807  #endif
808  }
809  }
810  _rx_response_length = 0U;
811  }
812  scheduler_resume(false);
813 #ifdef DEBUG_TASK_MEASURE
814  r2r_stop_task(_task_prev_id_callback);
815 #endif
