567 void _SIM908_callback(char data)
568 {
569  #ifdef PC_CALLBACK
570  uart1_send_char(data); /* Mirroring communication from sim908 to uart1 */
571  #endif
572 
573  _rx_response_length++;
574  _rx_buffer[_rx_buffer_tail = (_rx_buffer_tail + 1) % RX_BUFFER_SIZE] = data; /* Stores received data in buffer. This technically starts from index '1', but as the buffer is circular, it does not matter */
575 
576  if (data == CR) { /* Checking and counting for CR and LF */
577  _CR_counter++;
578  } else if (data == LF) {
579  _LF_counter++;
580  }
581 
582  if (_CR_counter > 0 && _LF_counter > 0) {
583  _CR_counter = _LF_counter = 0;
584  if (_check_response(SIM908_RESPONSE_CR_LF) || _check_response(SIM908_RESPONSE_LF_CR)) { /* Skipping empty lines */
585  } else if (_check_response(SIM908_RESPONSE_OK)) { /* OK */
586  _ack_response_flag = SIM908_FLAG_OK;
587  } else if (_check_response(SIM908_RESPONSE_ERROR)) { /* Error */
588  _ack_response_flag = SIM908_FLAG_ERROR;
589  } else if (_check_response(SIM908_RESPONSE_CREG)) { /* CREG */
590  EXT_CONNECTION_STATUS_FLAG = _char_at(7, _rx_buffer_tail, _rx_response_length) == '1' ? CREG_REGISTERED : CREG_NOT_REGISTERED;
591  } else if (_check_response(SIM908_RESPONSE_GPS_READY)) { /* GPS Ready */
592  _ack_gps_response_flag = SIM908_FLAG_GPS_OK;
593  } else if (_check_response(SIM908_RESPONSE_FTP_PUT)) { /* FTPPUT */
594  /* FTP PUT OPEN SESSION: "+FTPPUT:1,1,1260"
595  FTP PUT RESPONSE: "+FTPPUT:2,140"
596  FTP PUT CLOSE SESSION: "+FTPPUT:1,0" */
597  char c1 = _char_at(8, _rx_buffer_tail, _rx_response_length);
598  char c2 = _char_at(10, _rx_buffer_tail, _rx_response_length);
599  char c3 = _char_at(11, _rx_buffer_tail, _rx_response_length);
600  if (c1 == '1' && c2 == '1' && c3 == ',') {
601  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_OPEN;
602  } else if(c1 == '2' && c2 == '1' && c3 == '4') {
603  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_SUCCESS;
604  } else if(c1 == '1' && c2 == '0') {
605  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_CLOSE;
606  } else {
607  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_ERROR;
608  }
609  } else if (_check_response(SIM908_RESPONSE_GPS_PULL)) { /* GPS pull */
610  _gps_response_tail = _rx_buffer_tail;
611  _gps_response_length = _rx_response_length;
612  _ack_gps_response_flag = SIM908_FLAG_GPS_PULL;
613  } else if (_check_response(SIM908_RESPONSE_RDY)) { /* System ready */
614  _system_running_flag = SIM908_FLAG_RUNNING;
615  } else if (_check_response(SIM908_RESPONSE_AT)) { /* Sync AT cmd */
616  _rx_response_length = 0;
617  }
618  _rx_response_length = 0;
619  }
620 }
