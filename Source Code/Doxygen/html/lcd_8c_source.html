134 /*************************************************************************
135 Low-level function to write byte to LCD controller
136 Input: data byte to write to LCD
137  rs 1: write data
138  0: write instruction
139 Returns: none
140 *************************************************************************/
141 #if LCD_IO_MODE
142 static void lcd_write(uint8_t data,uint8_t rs)
143 {
144  unsigned char dataBits ;
145 
146 
147  if (rs) { /* write data (RS=1, RW=0) */
148  lcd_rs_high();
149  } else { /* write instruction (RS=0, RW=0) */
150  lcd_rs_low();
151  }
152  lcd_rw_low();
153 
154  if ( ( &LCD_DATA0_PORT == &LCD_DATA1_PORT) && ( &LCD_DATA1_PORT == &LCD_DATA2_PORT ) && ( &LCD_DATA2_PORT == &LCD_DATA3_PORT )
155  && (LCD_DATA0_PIN == 0) && (LCD_DATA1_PIN == 1) && (LCD_DATA2_PIN == 2) && (LCD_DATA3_PIN == 3) )
156  {
157  /* configure data pins as output */
158  DDR(LCD_DATA0_PORT) |= 0x0F;
159 
160  /* output high nibble first */
161  dataBits = LCD_DATA0_PORT & 0xF0;
162  LCD_DATA0_PORT = dataBits |((data>>4)&0x0F);
163  lcd_e_toggle();
164 
165  /* output low nibble */
166  LCD_DATA0_PORT = dataBits | (data&0x0F);
167  lcd_e_toggle();
168 
169  /* all data pins high (inactive) */
170  LCD_DATA0_PORT = dataBits | 0x0F;
171  }
172  else
173  {
174  /* configure data pins as output */
175  DDR(LCD_DATA0_PORT) |= _BV(LCD_DATA0_PIN);
176  DDR(LCD_DATA1_PORT) |= _BV(LCD_DATA1_PIN);
177  DDR(LCD_DATA2_PORT) |= _BV(LCD_DATA2_PIN);
178  DDR(LCD_DATA3_PORT) |= _BV(LCD_DATA3_PIN);
179 
180  /* output high nibble first */
181  LCD_DATA3_PORT &= ~_BV(LCD_DATA3_PIN);
182  LCD_DATA2_PORT &= ~_BV(LCD_DATA2_PIN);
183  LCD_DATA1_PORT &= ~_BV(LCD_DATA1_PIN);
184  LCD_DATA0_PORT &= ~_BV(LCD_DATA0_PIN);
185  if(data & 0x80) LCD_DATA3_PORT |= _BV(LCD_DATA3_PIN);
186  if(data & 0x40) LCD_DATA2_PORT |= _BV(LCD_DATA2_PIN);
187  if(data & 0x20) LCD_DATA1_PORT |= _BV(LCD_DATA1_PIN);
188  if(data & 0x10) LCD_DATA0_PORT |= _BV(LCD_DATA0_PIN);
189  lcd_e_toggle();
190 
191  /* output low nibble */
192  LCD_DATA3_PORT &= ~_BV(LCD_DATA3_PIN);
193  LCD_DATA2_PORT &= ~_BV(LCD_DATA2_PIN);
194  LCD_DATA1_PORT &= ~_BV(LCD_DATA1_PIN);
195  LCD_DATA0_PORT &= ~_BV(LCD_DATA0_PIN);
196  if(data & 0x08) LCD_DATA3_PORT |= _BV(LCD_DATA3_PIN);
197  if(data & 0x04) LCD_DATA2_PORT |= _BV(LCD_DATA2_PIN);
198  if(data & 0x02) LCD_DATA1_PORT |= _BV(LCD_DATA1_PIN);
199  if(data & 0x01) LCD_DATA0_PORT |= _BV(LCD_DATA0_PIN);
200  lcd_e_toggle();
201 
202  /* all data pins high (inactive) */
203  LCD_DATA0_PORT |= _BV(LCD_DATA0_PIN);
204  LCD_DATA1_PORT |= _BV(LCD_DATA1_PIN);
205  LCD_DATA2_PORT |= _BV(LCD_DATA2_PIN);
206  LCD_DATA3_PORT |= _BV(LCD_DATA3_PIN);
207  }
208 }
209 #else
210 #define lcd_write(d,rs) if (rs) *(volatile uint8_t*)(LCD_IO_DATA) = d; else *(volatile uint8_t*)(LCD_IO_FUNCTION) = d;
211 /* rs==0 -> write instruction to LCD_IO_FUNCTION */
212 /* rs==1 -> write data to LCD_IO_DATA */

216 /*************************************************************************
217 Low-level function to read byte from LCD controller
218 Input: rs 1: read data
219  0: read busy flag / address counter
220 Returns: byte read from LCD controller
221 *************************************************************************/
222 #if LCD_IO_MODE
223 static uint8_t lcd_read(uint8_t rs)
224 {
225  uint8_t data;
226 
227 
228  if (rs)
229  lcd_rs_high(); /* RS=1: read data */
230  else
231  lcd_rs_low(); /* RS=0: read busy flag */
232  lcd_rw_high(); /* RW=1 read mode */
233 
234  if ( ( &LCD_DATA0_PORT == &LCD_DATA1_PORT) && ( &LCD_DATA1_PORT == &LCD_DATA2_PORT ) && ( &LCD_DATA2_PORT == &LCD_DATA3_PORT )
235  && ( LCD_DATA0_PIN == 0 )&& (LCD_DATA1_PIN == 1) && (LCD_DATA2_PIN == 2) && (LCD_DATA3_PIN == 3) )
236  {
237  DDR(LCD_DATA0_PORT) &= 0xF0; /* configure data pins as input */
238 
239  lcd_e_high();
240  lcd_e_delay();
241  data = PIN(LCD_DATA0_PORT) << 4; /* read high nibble first */
242  lcd_e_low();
243 
244  lcd_e_delay(); /* Enable 500ns low */
245 
246  lcd_e_high();
247  lcd_e_delay();
248  data |= PIN(LCD_DATA0_PORT)&0x0F; /* read low nibble */
249  lcd_e_low();
250  }
251  else
252  {
253  /* configure data pins as input */
254  DDR(LCD_DATA0_PORT) &= ~_BV(LCD_DATA0_PIN);
255  DDR(LCD_DATA1_PORT) &= ~_BV(LCD_DATA1_PIN);
256  DDR(LCD_DATA2_PORT) &= ~_BV(LCD_DATA2_PIN);
257  DDR(LCD_DATA3_PORT) &= ~_BV(LCD_DATA3_PIN);
258 
259  /* read high nibble first */
260  lcd_e_high();
261  lcd_e_delay();
262  data = 0;
263  if ( PIN(LCD_DATA0_PORT) & _BV(LCD_DATA0_PIN) ) data |= 0x10;
264  if ( PIN(LCD_DATA1_PORT) & _BV(LCD_DATA1_PIN) ) data |= 0x20;
265  if ( PIN(LCD_DATA2_PORT) & _BV(LCD_DATA2_PIN) ) data |= 0x40;
266  if ( PIN(LCD_DATA3_PORT) & _BV(LCD_DATA3_PIN) ) data |= 0x80;
267  lcd_e_low();
268 
269  lcd_e_delay(); /* Enable 500ns low */
270 
271  /* read low nibble */
272  lcd_e_high();
273  lcd_e_delay();
274  if ( PIN(LCD_DATA0_PORT) & _BV(LCD_DATA0_PIN) ) data |= 0x01;
275  if ( PIN(LCD_DATA1_PORT) & _BV(LCD_DATA1_PIN) ) data |= 0x02;
276  if ( PIN(LCD_DATA2_PORT) & _BV(LCD_DATA2_PIN) ) data |= 0x04;
277  if ( PIN(LCD_DATA3_PORT) & _BV(LCD_DATA3_PIN) ) data |= 0x08;
278  lcd_e_low();
279  }
280  return data;
281 }
282 #else
283 #define lcd_read(rs) (rs) ? *(volatile uint8_t*)(LCD_IO_DATA+LCD_IO_READ) : *(volatile uint8_t*)(LCD_IO_FUNCTION+LCD_IO_READ)
284 /* rs==0 -> read instruction from LCD_IO_FUNCTION */
285 /* rs==1 -> read data from LCD_IO_DATA */
286 #endif

289 /*************************************************************************
290 loops while lcd is busy, returns address counter
291 *************************************************************************/
292 static uint8_t lcd_waitbusy(void)
293 
294 {
295  register uint8_t c;
296 
297  /* wait until busy flag is cleared */
298  while ( (c=lcd_read(0)) & (1<<LCD_BUSY)) {}
299 
300  /* the address counter is updated 4us after the busy flag is cleared */
301  delay(2);
302 
303  /* now read the address counter */
304  return (lcd_read(0)); // return address counter
305

439 /*************************************************************************
440 Display character at current cursor position
441 Input: character to be displayed
442 Returns: none
443 *************************************************************************/
444 void lcd_putc(char c)
445 {
446  uint8_t pos;
447 
448  pos = lcd_waitbusy(); // read busy-flag and address counter
449  if (c=='\n')
450  {
451  lcd_newline(pos);
452  }
453  else
454  {
455 #if LCD_WRAP_LINES==1
456 #if LCD_LINES==1
457  if ( pos == LCD_START_LINE1+LCD_DISP_LENGTH ) {
458  lcd_write((1<<LCD_DDRAM)+LCD_START_LINE1,0);
459  }
460 #elif LCD_LINES==2
461  if ( pos == LCD_START_LINE1+LCD_DISP_LENGTH ) {
462  lcd_write((1<<LCD_DDRAM)+LCD_START_LINE2,0);
463  }else if ( pos == LCD_START_LINE2+LCD_DISP_LENGTH ){
464  lcd_write((1<<LCD_DDRAM)+LCD_START_LINE1,0);
465  }
466 #elif LCD_LINES==4
467  if ( _last_pos == LCD_START_LINE1+LCD_DISP_LENGTH ) {
468  lcd_write((1<<LCD_DDRAM)+LCD_START_LINE2,0);
469  }else if ( _last_pos == LCD_START_LINE2+LCD_DISP_LENGTH - 1 ) {
470  lcd_write((1<<LCD_DDRAM)+LCD_START_LINE3,0);
471  }else if ( _last_pos == LCD_START_LINE3+LCD_DISP_LENGTH - 1 ) {
472  lcd_write((1<<LCD_DDRAM)+LCD_START_LINE4,0);
473  }else if ( _last_pos == LCD_START_LINE4+LCD_DISP_LENGTH - 1 ) {
474  lcd_write((1<<LCD_DDRAM)+LCD_START_LINE1,0);
475  }
476  _last_pos = pos;
477 #endif
478  lcd_waitbusy();
479 #endif
480  lcd_write(c, 1);
481  }
482 
483 }/* lcd_putc */

520 /*************************************************************************
521 Initialize display and select type of cursor
522 Input: dispAttr LCD_DISP_OFF display off
523  LCD_DISP_ON display on, cursor off
524  LCD_DISP_ON_CURSOR display on, cursor on
525  LCD_DISP_CURSOR_BLINK display on, cursor on flashing
526 Returns: none
527 *************************************************************************/
528 void lcd_init(uint8_t dispAttr)
529 {
530 #if LCD_IO_MODE
531  /*
532  * Initialize LCD to 4 bit I/O mode
533  */
534 
535  if ( ( &LCD_DATA0_PORT == &LCD_DATA1_PORT) && ( &LCD_DATA1_PORT == &LCD_DATA2_PORT ) && ( &LCD_DATA2_PORT == &LCD_DATA3_PORT )
536  && ( &LCD_RS_PORT == &LCD_DATA0_PORT) && ( &LCD_RW_PORT == &LCD_DATA0_PORT) && (&LCD_E_PORT == &LCD_DATA0_PORT)
537  && (LCD_DATA0_PIN == 0 ) && (LCD_DATA1_PIN == 1) && (LCD_DATA2_PIN == 2) && (LCD_DATA3_PIN == 3)
538  && (LCD_RS_PIN == 4 ) && (LCD_RW_PIN == 5) && (LCD_E_PIN == 6 ) )
539  {
540  /* configure all port bits as output (all LCD lines on same port) */
541  DDR(LCD_DATA0_PORT) |= 0x7F;
542  }
543  else if ( ( &LCD_DATA0_PORT == &LCD_DATA1_PORT) && ( &LCD_DATA1_PORT == &LCD_DATA2_PORT ) && ( &LCD_DATA2_PORT == &LCD_DATA3_PORT )
544  && (LCD_DATA0_PIN == 0 ) && (LCD_DATA1_PIN == 1) && (LCD_DATA2_PIN == 2) && (LCD_DATA3_PIN == 3) )
545  {
546  /* configure all port bits as output (all LCD data lines on same port, but control lines on different ports) */
547  DDR(LCD_DATA0_PORT) |= 0x0F;
548  DDR(LCD_RS_PORT) |= _BV(LCD_RS_PIN);
549  DDR(LCD_RW_PORT) |= _BV(LCD_RW_PIN);
550  DDR(LCD_E_PORT) |= _BV(LCD_E_PIN);
551  }
552  else
553  {
554  /* configure all port bits as output (LCD data and control lines on different ports */
555  DDR(LCD_RS_PORT) |= _BV(LCD_RS_PIN);
556  DDR(LCD_RW_PORT) |= _BV(LCD_RW_PIN);
557  DDR(LCD_E_PORT) |= _BV(LCD_E_PIN);
558  DDR(LCD_DATA0_PORT) |= _BV(LCD_DATA0_PIN);
559  DDR(LCD_DATA1_PORT) |= _BV(LCD_DATA1_PIN);
560  DDR(LCD_DATA2_PORT) |= _BV(LCD_DATA2_PIN);
561  DDR(LCD_DATA3_PORT) |= _BV(LCD_DATA3_PIN);
562  }
563  delay(16000); /* wait 16ms or more after power-on */
564 
565  /* initial write to lcd is 8bit */
566  LCD_DATA1_PORT |= _BV(LCD_DATA1_PIN); // _BV(LCD_FUNCTION)>>4;
567  LCD_DATA0_PORT |= _BV(LCD_DATA0_PIN); // _BV(LCD_FUNCTION_8BIT)>>4;
568  lcd_e_toggle();
569  delay(4992); /* delay, busy flag can't be checked here */
570 
571  /* repeat last command */
572  lcd_e_toggle();
573  delay(64); /* delay, busy flag can't be checked here */
574 
575  /* repeat last command a third time */
576  lcd_e_toggle();
577  delay(64); /* delay, busy flag can't be checked here */
578 
579  /* now configure for 4bit mode */
580  LCD_DATA0_PORT &= ~_BV(LCD_DATA0_PIN); // LCD_FUNCTION_4BIT_1LINE>>4
581  lcd_e_toggle();
582  delay(64); /* some displays need this additional delay */
583 
584  /* from now the LCD only accepts 4 bit I/O, we can use lcd_command() */
585 #else
586  /*
587  * Initialize LCD to 8 bit memory mapped mode
588  */
589 
590  /* enable external SRAM (memory mapped lcd) and one wait state */
591  MCUCR = _BV(SRE) | _BV(SRW);
592 
593  /* reset LCD */
594  delay(16000); /* wait 16ms after power-on */
595  lcd_write(LCD_FUNCTION_8BIT_1LINE,0); /* function set: 8bit interface */
596  delay(4992); /* wait 5ms */
