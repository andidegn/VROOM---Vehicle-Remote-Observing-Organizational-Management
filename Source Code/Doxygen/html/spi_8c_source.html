199

202

204  _cs_pin = __param->cs_pin;
205  _cs_active_level = __param->cs_active_level;
206  DDR_SPI |= _BV(_cs_pin);
207 
208  /* Sets an additional bit if the divider is 2, 8 or 32 */
209  if ((__param->freq_divider == SPI_DIVIDER_2) ||
210  (__param->freq_divider == SPI_DIVIDER_8) ||
211  (__param->freq_divider == SPI_DIVIDER_32)) {
212  SPSR |= _BV(SPI2X);
213  } else {
214  SPSR &= ~_BV(SPI2X);

217  /* Enables SPI, sets it as Master and sets all related bits (p.202)
218  Bit | 7 | 6 | 5 | 4 | 3 | 2 | 1 | 0 |
219  0x2C (0x4C) |SPIE| SPE|DORD|MSTR|CPOL|CPHA|SPR1|SPR0| SPCR
