565 void _SIM908_callback(char data)
566 {
567  #ifdef DEBUG_UART_ENABLE
568  uart1_send_char(data); /* Mirroring communication from sim908 to uart1 */
569  #endif
570 
571  _rx_response_length++;
572  _rx_buffer[_rx_buffer_tail = (_rx_buffer_tail + 1) % RX_BUFFER_SIZE] = data; /* Stores received data in buffer. This technically starts from index '1', but as the buffer is circular, it does not matter */
573 
574  if (data == CR) { /* Checking and counting for CR and LF */
575  _CR_counter++;
576  } else if (data == LF) {
577  _LF_counter++;
578  }
579 
580  if (_CR_counter > 0 && _LF_counter > 0) {
581  _CR_counter = _LF_counter = 0;
582  if (_check_response(SIM908_RESPONSE_CR_LF) || _check_response(SIM908_RESPONSE_LF_CR)) { /* Skipping empty lines */
583  } else if (_check_response(SIM908_RESPONSE_OK)) { /* OK */
584  _ack_response_flag = SIM908_FLAG_OK;
585  } else if (_check_response(SIM908_RESPONSE_ERROR)) { /* Error */
586  _ack_response_flag = SIM908_FLAG_ERROR;
587  } else if (_check_response(SIM908_RESPONSE_GPS_PULL)) { /* GPS pull */
588  if (_gps_pull_flag == SIM908_FLAG_GPS_PULL) {
589  _gps_response_tail = _rx_buffer_tail;
590  _gps_response_length = _rx_response_length;
591  _ack_gps_response_flag = SIM908_FLAG_GPS_PULL_OK;
592  _gps_pull_flag = SIM908_FLAG_WAITING;
593  }
594  } else if (_check_response(SIM908_RESPONSE_CREG)) { /* CREG */
595  EXT_CONNECTION_CREG_FLAG = _char_at(7, _rx_buffer_tail, _rx_response_length) == '1' ? CREG_REGISTERED : CREG_NOT_REGISTERED;
596  } else if (_check_response(SIM908_RESPONSE_GPS_READY)) { /* GPS Ready */
597  _ack_gps_response_flag = SIM908_FLAG_GPS_OK;
598  } else if (_check_response(SIM908_RESPONSE_FTP_PUT)) { /* FTPPUT */
599  /* FTP PUT OPEN SESSION: "+FTPPUT:1,1,1260"
600  FTP PUT RESPONSE: "+FTPPUT:2,140"
601  FTP PUT CLOSE SESSION: "+FTPPUT:1,0" */
602  char c1 = _char_at(8, _rx_buffer_tail, _rx_response_length);
603  char c2 = _char_at(10, _rx_buffer_tail, _rx_response_length);
604  char c3 = _char_at(11, _rx_buffer_tail, _rx_response_length);
605  if (c1 == '1' && c2 == '1' && c3 == ',') {
606  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_OPEN;
607  } else if(c1 == '2' && c2 == '1' && c3 == '4') {
608  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_SUCCESS;
609  } else if(c1 == '1' && c2 == '0') {
610  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_CLOSE;
611  } else {
612  _ack_ftp_response_flag = SIM908_FLAG_FTP_PUT_ERROR;
613  }
614  } else if (_check_response(SIM908_RESPONSE_RDY)) { /* System ready */
615  _system_running_flag = SIM908_FLAG_RUNNING;
616  } else if (_check_response(SIM908_RESPONSE_AT)) { /* Sync AT cmd */
617  _rx_response_length = 0;
618  }
619  _rx_response_length = 0;
620  }
621 }
